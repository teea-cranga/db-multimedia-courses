CREATE OR REPLACE PROCEDURE PS_INSERARE_IMG (V_ID IN NUMBER, V_DENUMIRE IN VARCHAR2, V_DENUMIRE_STIINTIFICA IN VARCHAR2, V_FISIER IN BLOB) IS
BEGIN
    INSERT INTO GANDACI VALUES(V_ID, V_DENUMIRE, V_DENUMIRE_STIINTIFICA, ORDImage(V_FISIER, 1), ORDImageSignature.INIT());
    COMMIT;
END;
/

-- proof it was inserted


SET SERVEROUTPUT ON;
DECLARE
    VB ORDIMAGE;
BEGIN
    SELECT IMG INTO VB FROM GANDACI WHERE ID=1;
    DBMS_OUTPUT.PUT_LINE('INALTIME IMAGINE: ' || VB.getHeight() || ', LATIME IMAGINE: ' || VB.getContentLength());
END;
